\begin{exercise}
Find the area of the polygon shown below.
  \begin{center}
\begin{tikzpicture}[scale=1]
\draw[thin,gray!40] (-3,-2) grid (3,4);
  \draw[<->] (-3,0)--(3,0);
  \draw[<->] (0,-2)--(0,4);
  \filldraw[blue, opacity=0.3](-2,-1)--(2,0)--(1,3)--(-1,2)--cycle;
 \end{tikzpicture}
\end{center}  

$$\text{Area}=\answer{9}\,\text{square units}$$
 \end{exercise}
